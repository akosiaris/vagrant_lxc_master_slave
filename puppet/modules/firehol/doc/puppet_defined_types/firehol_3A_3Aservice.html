define firehol::service (
  String $service_name = $title,
  String $server       = '',
  String $client       = 'default',
  String $config_file  = '/etc/firehol/firehol.conf',
) {

  concat::fragment { "service_${service_name}":
    target  => $config_file,
    content => template('firehol/firehol_service.erb'),
    order   => 010,
  }

}
